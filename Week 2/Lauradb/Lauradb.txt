Table users {
  id integer [primary key]
  username varchar(50)
}

Table hero {
  id integer
  name varchar(50)
  description1 text
  description2 text
}

Table aboutme {
  id integer
  name varchar(50)
  age integer
  website varchar(50)
  degree varchar(50)
  phone varchar(50)
  phemailone varchar(50)
  city varchar(50)
  freelance varchar(50)
}

Table skill {
  id integer
  name varchar(50)
  count_skill integer
  description text
}

Table code {
  id integer
  name varchar(50)
  count_code integer
}

Table sumary {
  id integer
  name varchar(50)
  year varchar(50)
  description text
  details text
}

Table education {
  id integer
  name varchar(50)
  year varchar(50)
  description text
  details text
}

Table experience {
  id integer
  name varchar(50)
  year varchar(50)
  description text
  details text
}

Table myservices {
  id integer
  name varchar(50)
  description text
}

Table testimonials {
  id integer
  name varchar(50)
  profile_image varchar(50)
  job varchar(50)
  description text
}

Table myportfolio {
  id integer
  portfolio_id integer [primary key]
  name varchar(50)
  portfolio_link_img varchar(50)
  portfolio_link varchar(50) [primary key]
  portfolio_type varchar(50)
}

Table portfolioimg {
  id integer
  portfolio_id integer
  name varchar(50)
  portfolio_link_img varchar(50) [primary key]
}

Table portfoliodetails {
  id integer
  portfolio_id integer
  name varchar(50)
  category varchar(50)
  client varchar(50)
  portfolio_date timestamp
  portfolio_link varchar(50)
  description text
}

Table pricing {
  id integer
  name varchar(50)
  price float
  description text
  pay_link varchar(50)
}

Table sosial {
  id integer
  sosial_name varchar(50)
  sosial_link varchar(50)
}

Table contact {
  id integer
  name varchar(50)
  email varchar(50)
  subject varchar(50)
  message text
}
